# 二分查找
```c
#include <stdio.h>
int search(int* nums, int numsSize, int target) { 
	int left = 0; 
	int right = numsSize - 1; 
	while(left <= right){ 
		int i = left + (right - left) / 2; 
		if(nums[i] < target){ 
			left = i + 1; 
		}else if(nums[i] > target){ 
			right = i - 1; 
		}else{ 
			return i;} 
		} 
	return -1;
} 
int main() { 
	int numsSize; // 数组长度 
	int target; // 要查找的目标值 
	printf("请输入数组的长度："); 
	scanf("%d", &numsSize); 
	int nums[numsSize]; 
	printf("请输入 %d 个有序的整数（以空格分隔）：\n", numsSize); 
	for (int i = 0; i < numsSize; i++) { 
		scanf("%d", &nums[i]); 
	} 
	// 3. 输入要查找的目标值 printf("请输入要查找的目标值："); 
	scanf("%d", &target); 
	// 4. 调用二分查找函数 
	int result = search(nums, numsSize, target); 
	// 5. 输出结果 
	if (result != -1) { 
		printf("目标值 %d 在数组中的下标是：%d\n", target, result); 
	} else { 
		printf("数组中未找到目标值 %d\n", target); 
	} 
	return 0; 
}
```
# 移除元素
```c
int removeElement(int* nums, int numsSize, int val){
    int slow = 0;
    for(int fast = 0; fast < numsSize; fast++) {
        //若快指针位置的元素不等于要删除的元素
        if(nums[fast] != val) {
            //将其挪到慢指针指向的位置，慢指针+1
            nums[slow++] = nums[fast];
        }
    }
    //最后慢指针的大小就是新的数组的大小
    return slow;
}
```

# 有序数组的平方
有几个要注意的点，和`cpp`不一样
`int *result = (int*)malloc(numsSize * sizeof(int));`这样子声明数组才是声明在堆上，
如果直接使用`int result[numsSize]`不对，这样是声明在栈上的局部变量，在函数结束时会被清除，所以函数返回一定是空的
还有要注意的，和`cpp`不一样的地方在于，需要维护一个`newNumsSize`为了给
```c
#include <stdio.h>
int *sortSquare(int *nums, int numsSize, int *newNumsSize){
	*newNumsSize = numsSize;
	int left = 0;
	int right = numsSize - 1;
	int *result = (int*)malloc(numsSize * sizeof(int));
	int index = numsSize - 1;
	while(left <= right){
		if(nums[left] * nums[left] <= nums[right] * nums[right]){
			result[index] = nums[right] * nums[right];
			right--;
			index--;
		}else{
			result[index] = nums[left] * nums[left];
			left++;
			index--;
		}
	}
	return result;
}

int main(){
	int n;
	scanf("%d", &n);
	int nums[n];
	int i;
	for(i = 0; i < n; i++){
		scanf("%d", &nums[i]);
	}
	int newNumsSize;
	int *result = sortSquare(&nums, n, &newNumsSize);
	for(i = 0; i < newNumsSize; i++){
		printf("%d", result[i]);
	}
	
	return 0;
}
```
